01201 /*
01202  * xQueueAltGenericSend() is an alternative version of xQueueGenericSend().
01203  * Likewise xQueueAltGenericReceive() is an alternative version of
01204  * xQueueGenericReceive().
01205  *
01206  * The source code that implements the alternative (Alt) API is much
01207  * simpler  because it executes everything from within a critical section.
01208  * This is  the approach taken by many other RTOSes, but FreeRTOS.org has the
01209  * preferred fully featured API too.  The fully featured API has more
01210  * complex  code that takes longer to execute, but makes much less use of
01211  * critical sections.  Therefore the alternative API sacrifices interrupt
01212  * responsiveness to gain execution speed, whereas the fully featured API
01213  * sacrifices execution speed to ensure better interrupt responsiveness.
01214  */
01215 signed portBASE_TYPE xQueueAltGenericSend( xQueueHandle pxQueue, const void * const pvItemToQueue, portTickType xTicksToWait, portBASE_TYPE xCopyPosition );
01216 signed portBASE_TYPE xQueueAltGenericReceive( xQueueHandle pxQueue, void * const pvBuffer, portTickType xTicksToWait, portBASE_TYPE xJustPeeking );
01217 #define xQueueAltSendToFront( xQueue, pvItemToQueue, xTicksToWait ) xQueueAltGenericSend( ( xQueue ), ( pvItemToQueue ), ( xTicksToWait ), queueSEND_TO_FRONT )
01218 #define xQueueAltSendToBack( xQueue, pvItemToQueue, xTicksToWait ) xQueueAltGenericSend( ( xQueue ), ( pvItemToQueue ), ( xTicksToWait ), queueSEND_TO_BACK )
