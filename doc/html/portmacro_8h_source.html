00121 /* 
00122  * Set basepri to portMAX_SYSCALL_INTERRUPT_PRIORITY without effecting other
00123  * registers.  r0 is clobbered.
00124  */ 
00125 #define portSET_INTERRUPT_MASK()                        \
00126     __asm volatile                                      \
00127     (                                                   \
00128         "   mov r0, %0                              \n" \
00129         "   msr basepri, r0                         \n" \
00130         ::"i"(configMAX_SYSCALL_INTERRUPT_PRIORITY):"r0"    \
00131     )
00132     
00133 /*
00134  * Set basepri back to 0 without effective other registers.
00135  * r0 is clobbered.  FAQ:  Setting BASEPRI to 0 is not a bug.  Please see 
00136  * http://www.FreeRTOS.org/RTOS-Cortex-M3-M4.html before disagreeing.
00137  */
00138 #define portCLEAR_INTERRUPT_MASK()          \
00139     __asm volatile                          \
00140     (                                       \
00141         "   mov r0, #0                  \n" \
00142         "   msr basepri, r0             \n" \
00143         :::"r0"                             \
00144     )
00145 
00146 /* FAQ:  Setting BASEPRI to 0 in portCLEAR_INTERRUPT_MASK_FROM_ISR() is not a 
00147 bug.  Please see http://www.FreeRTOS.org/RTOS-Cortex-M3-M4.html before 
00148 disagreeing. */
00149 #define portSET_INTERRUPT_MASK_FROM_ISR()       0;portSET_INTERRUPT_MASK()
00150 #define portCLEAR_INTERRUPT_MASK_FROM_ISR(x)    portCLEAR_INTERRUPT_MASK();(void)x
